388 inline bool almostEqualRelativeAndAbs(double a,
389  double b,
390  double max_diff = 1e-6,
391  double max_rel_diff = std::numeric_limits<double>::epsilon())
392 {
393  double diff = std::fabs(a - b);
394  if (diff <= max_diff)
395  return true;
396 
397  a = std::fabs(a);
398  b = std::fabs(b);
399  double largest = (b > a) ? b : a;
400 
401  return (diff <= largest * max_rel_diff);
402 }

416 inline bool almostEqualRelativeAndAbs(const Eigen::Ref<const Eigen::VectorXd>& v1,
417  const Eigen::Ref<const Eigen::VectorXd>& v2,
418  double max_diff,
419  double max_rel_diff = std::numeric_limits<double>::epsilon())
420 {
421  if (v1.size() == 0 && v2.size() == 0)
422  return true;
423  if (v1.size() != v2.size())
424  return false;
425 
426  Eigen::ArrayWrapper<const Eigen::Ref<const Eigen::VectorXd>> a1 = v1.array();
427  Eigen::ArrayWrapper<const Eigen::Ref<const Eigen::VectorXd>> a2 = v2.array();
428 
429  auto diff_abs = (a1 - a2).abs();
430  double diff = diff_abs.maxCoeff();
431  if (diff <= max_diff)
432  return true;
433 
434  return (diff_abs <= (max_rel_diff * a1.abs().max(a2.abs()))).all();
435 }
